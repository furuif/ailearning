## 一、正合新股（第一类）

### 筛选标准

```python
def is_zhenghe(stock_info):
    """
    第一类：正合新股判断标准
    
    满足以下至少 2 项：
    1. 热门题材（半导体/AI/医药/新能源/信创）
    2. 营收增速 > 20%
    3. 行业龙头或细分龙头
    4. 机构调研热度高
    5. 发行市盈率高于行业平均
    """
    score = 0
    
    # 题材判断
    if stock_info['industry'] in ['半导体', '人工智能', '创新药', '新能源', '信创']:
        score += 1
    
    # 增速判断
    if stock_info['revenue_growth'] > 20:
        score += 1
    
    # 行业地位
    if stock_info['is_leader']:
        score += 1
    
    # 机构调研
    if stock_info['institution_survey'] > 10:
        score += 1
    
    # 发行 PE
    if stock_info['issue_pe'] > stock_info['industry_pe']:
        score += 1
    
    return score >= 2
```

## 二、奇胜新股（第二类）

### 筛选标准

```python
def is_qisheng(stock_info):
    """
    第二类：奇胜新股判断标准
    
    满足以下至少 1 项：
    1. 名字有好彩头（带金/鑫/隆/泰/达等）
    2. 发行价有吉利数字（6/8/9 结尾）
    3. 发行价是特殊数字（如 11.11）
    4. 名字简洁好记（2-3 个字）
    """
    # 名字判断
    lucky_chars = ['金', '鑫', '隆', '泰', '达', '发', '富', '贵', '红', '宏']
    if any(c in stock_info['name'] for c in lucky_chars):
        return True
    
    # 发行价判断
    price_str = str(stock_info['issue_price'])
    if price_str.endswith('6') or price_str.endswith('8') or price_str.endswith('9'):
        return True
    
    # 特殊数字
    if '11.11' in price_str or '16.88' in price_str or '18.88' in price_str:
        return True
    
    # 简洁好记
    if len(stock_info['name']) <= 3:
        return True
    
    return False
```

```python
def is_putong(stock_info):
```

## 📝 分类实战流程

### 步骤一：获取数据

```python
import akshare as ak
import pandas as pd

# 获取新股数据
ipo_df = ak.stock_new_gh_cninfo()

# 筛选 2024 年
ipo_df['上市日期'] = pd.to_datetime(ipo_df['上市日期'])
ipo_2024 = ipo_df[(ipo_df['上市日期'].dt.year == 2024)]

# 保存原始数据
ipo_2024.to_csv('2024 年新股原始数据.csv', index=False, encoding='utf-8-sig')
```

### 步骤二：分类判断

```python
def classify_stock(row):
    """单只股票分类"""
    if is_zhenghe(row):
        return '正合'
    elif is_qisheng(row):
        return '奇胜'
    else:
        return '普通'

# 应用分类
ipo_2024['分类'] = ipo_2024.apply(classify_stock, axis=1)

# 统计
print(ipo_2024['分类'].value_counts())
```

### 步骤三：导出结果

```python
# 分类导出
zhenghe = ipo_2024[ipo_2024['分类'] == '正合']
qisheng = ipo_2024[ipo_2024['分类'] == '奇胜']
putong = ipo_2024[ipo_2024['分类'] == '普通']

# 保存
```
